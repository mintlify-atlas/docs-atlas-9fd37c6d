## Core types

### DateData

Object representing a date with parsed components.

```typescript
interface DateData {
  year: number;
  month: number;
  day: number;
  timestamp: number;
  dateString: string;
}
```

<ResponseField name="year" type="number">
  Full year (e.g., 2024)
</ResponseField>

<ResponseField name="month" type="number">
  Month number (1-12, where 1 = January)
</ResponseField>

<ResponseField name="day" type="number">
  Day of month (1-31)
</ResponseField>

<ResponseField name="timestamp" type="number">
  Unix timestamp in milliseconds
</ResponseField>

<ResponseField name="dateString" type="string">
  Date in 'yyyy-MM-dd' format
</ResponseField>

**Example:**

```typescript
const dateData: DateData = {
  year: 2024,
  month: 3,
  day: 15,
  timestamp: 1710460800000,
  dateString: '2024-03-15'
};
```

### DayState

String literal type for day state.

```typescript
type DayState = 'selected' | 'disabled' | 'inactive' | 'today' | '';
```

<ResponseField name="selected" type="'selected'">
  Day is currently selected
</ResponseField>

<ResponseField name="disabled" type="'disabled'">
  Day is disabled and cannot be selected
</ResponseField>

<ResponseField name="inactive" type="'inactive'">
  Day is outside the current month
</ResponseField>

<ResponseField name="today" type="'today'">
  Day is today's date
</ResponseField>

<ResponseField name="empty" type="''">
  Default state (empty string)
</ResponseField>

### Direction

Direction for calendar navigation.

```typescript
type Direction = 'left' | 'right';
```

### MarkingTypes

Available marking types for calendar days.

```typescript
type MarkingTypes = 'dot' | 'multi-dot' | 'period' | 'multi-period' | 'custom';
```

<ResponseField name="dot" type="'dot'">
  Single colored dot below the date
</ResponseField>

<ResponseField name="multi-dot" type="'multi-dot'">
  Multiple colored dots below the date
</ResponseField>

<ResponseField name="period" type="'period'">
  Background color spanning a date range
</ResponseField>

<ResponseField name="multi-period" type="'multi-period'">
  Multiple overlapping period markings
</ResponseField>

<ResponseField name="custom" type="'custom'">
  Custom marking component
</ResponseField>

### MarkedDates

Object mapping date strings to their marking configurations.

```typescript
type MarkedDates = {
  [key: string]: MarkingProps;
};
```

**Example:**

```typescript
const markedDates: MarkedDates = {
  '2024-03-15': {
    selected: true,
    selectedColor: '#4A90E2'
  },
  '2024-03-16': {
    marked: true,
    dotColor: 'red'
  },
  '2024-03-17': {
    disabled: true,
    disableTouchEvent: true
  }
};
```

## Agenda types

### AgendaEntry

Represents a single agenda item.

```typescript
interface AgendaEntry {
  name: string;
  height: number;
  day: string;
}
```

<ResponseField name="name" type="string">
  Name or title of the agenda item
</ResponseField>

<ResponseField name="height" type="number">
  Height of the item in pixels
</ResponseField>

<ResponseField name="day" type="string">
  Date string in 'yyyy-MM-dd' format
</ResponseField>

**Example:**

```typescript
const entry: AgendaEntry = {
  name: 'Team Meeting',
  height: 80,
  day: '2024-03-15'
};
```

### AgendaSchedule

Mapping of dates to agenda entries.

```typescript
type AgendaSchedule = {
```

```typescript
    { name: 'Conference', height: 100, day: '2024-03-16' }

};
```

## Usage with TypeScript

### Typed component props

```typescript
import React from 'react';
import { Calendar, DateData, MarkedDates } from 'react-native-calendars';

interface MyCalendarProps {
  initialDate: string;
  onDateSelect: (date: DateData) => void;
}

const MyCalendar: React.FC<MyCalendarProps> = ({
  initialDate,
  onDateSelect
}) => {
  const markedDates: MarkedDates = {
    [initialDate]: { selected: true }
  };

  return (
    <Calendar
      current={initialDate}
      onDayPress={onDateSelect}
      markedDates={markedDates}
    />
  );
};
```

### Typed context usage

```typescript
import React, { useContext } from 'react';
import { CalendarContext, CalendarContextProps } from 'react-native-calendars';

const MyComponent: React.FC = () => {
  const context = useContext<CalendarContextProps>(CalendarContext);

  const handlePress = () => {
    context.setDate('2024-03-15', 'dayPress');
  };

  return (
    <Text onPress={handlePress}>
      Current: {context.date}
    </Text>
  );
};
```
